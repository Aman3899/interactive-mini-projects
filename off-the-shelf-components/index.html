const progressBarFill = document.getElementById('progress-bar-fill');
const startProgressBtn = document.getElementById('start-progress-btn');
let progressInterval;

function startProgress() {
    let width = 0;
    progressBarFill.style.width = '0%'; // Reset
    clearInterval(progressInterval); // Clear existing interval if any

    progressInterval = setInterval(() => {
        if (width >= 100) {
            clearInterval(progressInterval);
            // Optional: Reset or indicate completion
            setTimeout(() => { progressBarFill.style.width = '0%'; }, 1000);
        } else {
            width++;
            progressBarFill.style.width = width + '%';
        }
    }, 30); // Adjust speed (milliseconds)
}

startProgressBtn?.addEventListener('click', startProgress);

     background-color: var(--step-complete-bg);

    animation: stepFadeIn 0.5s ease-in-out;

    to { opacity: 1; transform: translateY(0); }

/* Ensure global theme variables like --step-default-bg, --step-active-bg etc. are defined */
                        
const prevBtn = document.getElementById('prev-step-btn');
const nextBtn = document.getElementById('next-step-btn');
const stepIndicators = document.querySelectorAll('.step-indicator');
const stepContents = document.querySelectorAll('.step-content');
const progressLine = document.getElementById('step-progress-line');
const totalSteps = stepIndicators.length;
let currentStep = 1;

function updateStepForm() {
    // Update Indicators
    stepIndicators.forEach((indicator, index) => {
        indicator.classList.remove('active', 'completed');
        const step = parseInt(indicator.dataset.step);
        if (step === currentStep) {
            indicator.classList.add('active');
        } else if (step < currentStep) {
            indicator.classList.add('completed');
        }
    });

    // Update Content Visibility
    stepContents.forEach(content => {
        content.classList.add('hidden');
        if (parseInt(content.dataset.stepContent) === currentStep) {
            content.classList.remove('hidden');
        }
    });

    // Update Buttons State
    prevBtn.disabled = currentStep === 1;
    nextBtn.disabled = currentStep === totalSteps;
    // Change Next button text on last step (optional)
    nextBtn.textContent = currentStep === totalSteps ? 'Finish' : 'Next';


    // Update Progress Line Width
    const progressPercentage = ((currentStep - 1) / (totalSteps - 1)) * 100;
    progressLine.style.width = `${progressPercentage}%`;
}

prevBtn?.addEventListener('click', () => {
    if (currentStep > 1) {
        currentStep--;
        updateStepForm();
    }
});

nextBtn?.addEventListener('click', () => {
    if (currentStep < totalSteps) {
        currentStep++;
        updateStepForm();
    } else {
        // Handle form submission or final action
        console.log('Form Finished!');
         // Maybe show a confirmation inside the last step
    }
});

// Initial setup
updateStepForm();

/* Custom CSS for Rating (Theme variables handle colors) */
.rating-stars i {
    color: var(--rating-inactive-color); /* From theme */
    cursor: pointer;
    transition: color 0.2s ease, transform 0.2s ease;
    /* Using Font Awesome classes fa-regular fa-star / fa-solid fa-star */
}
/* Class applied on hover */
.rating-stars i:hover,
.rating-stars i.hover {
    color: var(--rating-active-color); /* From theme */
    transform: scale(1.2); /* Pop effect */
}
/* Class applied on click/active state */
.rating-stars i.active {
    color: var(--rating-active-color); /* From theme */
    /* Switches to solid star in JS */
}
/* Ensure global theme variables like --rating-inactive-color, --rating-active-color are defined */
                        
const ratingContainers = document.querySelectorAll('.rating-stars');
const ratingFeedback = document.getElementById('rating-feedback');

ratingContainers.forEach(container => {
    const stars = container.querySelectorAll('i');
    let currentRating = parseInt(container.dataset.rating) || 0;

    function updateStars(rating) {
        stars.forEach(star => {
            const starValue = parseInt(star.dataset.value);
            if (starValue <= rating) {
                star.classList.remove('fa-regular', 'hover'); // Remove hover if needed
                star.classList.add('fa-solid', 'active');
            } else {
                star.classList.remove('fa-solid', 'active', 'hover');
                star.classList.add('fa-regular');
            }
        });
    }

    stars.forEach(star => {
        star.addEventListener('mouseover', () => {
            const hoverValue = parseInt(star.dataset.value);
            stars.forEach(s => {
                const sValue = parseInt(s.dataset.value);
                if (sValue <= hoverValue) {
                    s.classList.add('hover', 'fa-solid'); // Show solid on hover temporarily
                    s.classList.remove('fa-regular');
                } else {
                     s.classList.remove('hover', 'fa-solid');
                     if (!s.classList.contains('active')) { // Don't remove regular if not active
                         s.classList.add('fa-regular');
                     }
                }
                 // Ensure active color is applied based on hover/active state
                s.style.color = (s.classList.contains('hover') || s.classList.contains('active')) ? 'var(--rating-active-color)' : 'var(--rating-inactive-color)';
            });
        });

        star.addEventListener('mouseout', () => {
            updateStars(currentRating); // Revert to current permanent rating
        });

        star.addEventListener('click', () => {
            currentRating = parseInt(star.dataset.value);
            container.dataset.rating = currentRating;
            updateStars(currentRating);
            if(ratingFeedback) ratingFeedback.textContent = `You rated ${currentRating} star(s)!`;
            // You might want to send this rating somewhere
        });
    });

    // Initialize display
    updateStars(currentRating);
});
